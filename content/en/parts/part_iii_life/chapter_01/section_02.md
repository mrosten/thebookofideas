<script>
function setNitrogenState(mode) {
    const n2State = document.getElementById('nitrogen-state');
    const tntState = document.getElementById('tnt-state');
    const btns = document.querySelectorAll('.nitrogen-visual .toggle-btn');
    const log = document.getElementById('reaction-text');

    btns.forEach(b => b.classList.remove('active'));

    if (mode === 'stable') {
        n2State.classList.add('active');
        tntState.classList.remove('active');
        btns[0].classList.add('active');
        log.textContent = "Status: Inert. The family unit is self-contained.";
        log.style.color = "#4cc9f0";
    } else {
        n2State.classList.remove('active');
        tntState.classList.add('active');
        btns[1].classList.add('active');
        log.textContent = "Status: CRITICAL. Seeking external validation triggers reactivity.";
        log.style.color = "#f72585";
    }
}
</script>
