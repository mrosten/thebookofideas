<script>
(function() {
    // Spine Logic
    const chain = document.getElementById('spineChain');
    if (chain) {
        for (let i = 0; i < 18; i++) {
            const v = document.createElement('div');
            v.className = 'vertebra';
            chain.appendChild(v);
        }
    }

    window.igniteSpine = function() {
        const verts = document.querySelectorAll('.vertebra');
        verts.forEach((v, i) => {
            setTimeout(() => {
                v.classList.add('active');
                setTimeout(() => v.classList.remove('active'), 800);
            }, i * 80);
        });
    };

    // Rebound Logic
    window.fireKarmic = function(type) {
        const orb = document.getElementById('energyOrb');
        const readout = document.getElementById('rebound-readout');
        
        orb.style.animation = 'none';
        orb.offsetHeight; // force reflow
        
        if (type === 'bless') {
            orb.style.background = 'radial-gradient(circle, #34d399 0%, #059669 100%)';
            orb.style.boxShadow = '0 0 20px #10b981';
            orb.style.animation = 'orb-bless 1.2s forwards cubic-bezier(0.4, 0, 0.2, 1)';
            readout.innerHTML = "<span style='color:#34d399'>TRANSMISSION:</span> Blessing accepted. Channel clear.";
        } else {
            orb.style.background = 'radial-gradient(circle, #f87171 0%, #b91c1c 100%)';
            orb.style.boxShadow = '0 0 20px #ef4444';
            orb.style.animation = 'orb-curse 1.8s forwards ease-in-out';
            setTimeout(() => {
                readout.innerHTML = "<span style='color:#f87171'>REJECTION:</span> Malice returned to source.";
            }, 900);
            readout.innerText = "Processing intent...";
        }
    };
})();
</script>
